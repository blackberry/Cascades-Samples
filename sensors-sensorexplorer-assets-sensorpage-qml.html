    import bb.cascades 1.0

    Page {
        Container {
            background: Color.DarkGray
            leftPadding: 30
            topPadding: 30
            rightPadding: 30
            bottomPadding: 30

            Label {
                horizontalAlignment: HorizontalAlignment.Center
                text: qsTr ("Sensor Properties")
                textStyle {
                    base: SystemDefaults.TextStyles.BigText
                    color: Color.White
                    fontWeight: FontWeight.Bold
                }
            }

            ListView {
                horizontalAlignment: HorizontalAlignment.Fill
                layoutProperties: StackLayoutProperties {
                    spaceQuota: 1
                }

                dataModel: _sensorExplorer.sensorPropertiesModel

                listItemComponents: [
                    ListItemComponent {
                        type: "item"
                        StandardListItem {
                            title: ListItemData.name
                            status: ListItemData.type
                            description: ListItemData.value
                        }
                    }
                ]
            }

            Container {
                horizontalAlignment: HorizontalAlignment.Center
                layout: StackLayout {
                    orientation: LayoutOrientation.LeftToRight
                }

                Button {
                    text: _sensorExplorer.sensorActive ? qsTr ("Stop") : qsTr ("Start")
                    onClicked: _sensorExplorer.sensorActive = !_sensorExplorer.sensorActive
                }

                Button {
                    text: qsTr ("Values")

                    onClicked: {
                        navigationPane.push(readingPage.createObject());
                    }
                    attachedObjects: [
                        ComponentDefinition {
                            id: readingPage
                            source: "ReadingPage.qml"
                        }
                    ]
                }
            }
        }
    }
